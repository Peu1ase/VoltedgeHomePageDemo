// ===== VoltEdge cart shared logic =====
const VOLT_CART_KEY = 'voltedge_cart';

function getCart(){
  try{ return JSON.parse(localStorage.getItem(VOLT_CART_KEY)) || []; }
  catch(e){ return []; }
}
function saveCart(cart){
  localStorage.setItem(VOLT_CART_KEY, JSON.stringify(cart));
}
function addToCart(item){
  const cart = getCart();
  const found = cart.find(c => c.id === item.id && c.color === item.color);
  if(found){ found.amount += 1; }
  else { cart.push({...item, amount:1}); }
  saveCart(cart);
}
function removeFromCart(index){
  const cart = getCart();
  cart.splice(index,1);
  saveCart(cart);
}
function cartTotal(){
  return getCart().reduce((sum,i)=> sum + i.price * i.amount, 0);
}
